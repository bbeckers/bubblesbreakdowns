 S = systempriors(M)

 \begin{itemize}
 \item
   \texttt{M} {[} model {]} - Model object on whose system properties the
   priors will be imposed.
 \end{itemize}
 
 \paragraph{Output arguments}
 
 \begin{itemize}
 \item
   \texttt{S} {[} systempriors {]} - New, empty system priors object.
 \end{itemize}
