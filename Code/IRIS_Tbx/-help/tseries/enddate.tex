

    \filetitle{enddate}{Date of the last available observation in a tseries object}{tseries/enddate}

	\paragraph{Syntax}
 
 \begin{verbatim}
 d = enddate(x)
 \end{verbatim}
 
 \paragraph{Input arguments}
 
 \begin{itemize}
 \item
   \texttt{x} {[} tseries {]} - Tseries object.
 \end{itemize}
 
 \paragraph{Output arguments}
 
 \begin{itemize}
 \item
   \texttt{d} {[} numeric {]} - IRIS serial date number representing the
   date of the last observation available in the input tseries.
 \end{itemize}
 
 \paragraph{Description}
 
 The \texttt{startdate} function is equivalent to calling
 
 \begin{verbatim}
 get(x,'endDate')
 \end{verbatim}
