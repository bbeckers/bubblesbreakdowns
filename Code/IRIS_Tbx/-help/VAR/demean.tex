

    \filetitle{demean}{Remove constant from VAR object}{VAR/demean}

	\paragraph{Syntax}
 
 \begin{verbatim}
 V = demean(V)
 \end{verbatim}
 
 \paragraph{Input arguments}
 
 \begin{itemize}
 \item
   \texttt{V} {[} VAR {]} - VAR object in which the constant vector will
   be reset to zero.
 \end{itemize}
 
 \paragraph{Output arguments}
 
 \begin{itemize}
 \item
   \texttt{V} {[} VAR {]} - VAR object with the constant vector reset to
   zero.
 \end{itemize}
