

    \filetitle{datboy}{Beginning of year for the specified daily date}{dates/datboy}

	\paragraph{Syntax}
 
 \begin{verbatim}
 Boy = dateboy(D)
 \end{verbatim}
 
 \paragraph{Input arguments}
 
 \begin{itemize}
 \item
   \texttt{D} {[} numeric {]} - Daily serial date number.
 \end{itemize}
 
 \paragraph{Output arguments}
 
 \begin{itemize}
 \item
   \texttt{Boy} {[} numeric {]} - Daily serial date number for the first
   day of the same year as \texttt{D}.
 \end{itemize}
