

    \filetitle{bb}{IRIS serial date numbers for dates with bi-monthly frequency}{dates/bb}

	\paragraph{Syntax}
 
 \begin{verbatim}
 d = bb(y)
 d = bb(y,b)
 \end{verbatim}
 
 \paragraph{Input arguments}
 
 \begin{itemize}
 \item
   \texttt{y} {[} numeric {]} - Years.
 \item
   \texttt{q} {[} numeric {]} - B-months; if missing, first bi-month is
   assumed.
 \end{itemize}
 
 \paragraph{Output arguments}
 
 \begin{itemize}
 \item
   \texttt{d} {[} numeric {]} - IRIS serial date numbers representing the
   input bi-months.
 \end{itemize}
